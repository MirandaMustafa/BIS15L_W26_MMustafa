wealth_gap <- gapminder %>%
  filter(year %in% c(1925, 2025)) %>%
  group_by(year) %>%
  mutate(gdp_percentile = ntile(gdp, 10)) %>% 
  filter(gdp_percentile %in% c(1, 10)) %>%
  mutate(wealth_group = ifelse(gdp_percentile == 1, "Poorest 10%", "Richest 10%")) %>%
  group_by(year, wealth_group) %>%
  summarize(avg_lex = mean(lex, na.rm = TRUE), .groups = 'drop') %>%
  pivot_wider(names_from = year, values_from = avg_lex)
